public static int pizza(double x, int n){
int k = (int)x + 1;
int p = n/(k+1), r = n%(k+1);
// ans - number of triangles
int ans = -1; // r=1 forbidden state
if( r != 1){
double t = (x*p + r - 1)/((x+1)*p + r);
if (t < x/(x+1)){
ans = 1;// the formula is correct
}
else{
ans = 0;// the formula is incorrect
}
}
return ans;
